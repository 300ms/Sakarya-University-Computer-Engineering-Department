#ifndef Merhabastack_hpp
#define Merhabastack_hpp
#include<iostream>
using namespace std;
class Merhabastack
{
public:
	int*staff;
	int stackHead;
	int staffN;
	int capacity;//Dizinin alabileceði max eleman sayýsýný gösterecektir.Yer ac cagirildiginde
	//varolan limitin iki katý kadar yer acilacaktir.
	bool isFull()
	{
		return staffN == capacity;
	}
	void yerAc(int boyut)
	{
		int*tmp = new int[boyut];
		for (int i = 0; i < staffN; i++)
			tmp[i] = staff[i];
		if (staff !=NULL)
			delete[] staff;
		staff = tmp;
		capacity = boyut;
	}
	Merhabastack()//Yeni bir nesne olusturulduðunda dizi  göstericisi NULL'u gösterecek sekilde nesneyi olusturur.
	{
		staff = NULL;
		capacity = 0;
		staffN = 0;
		stackHead = -1;
	}
	bool isEmpty()
	{
		return stackHead == -1;
	}
	void push(const int&newStaff)
	{
		if (isFull())
		{
			yerAc(2 *capacity+1);
		}
		stackHead++;
		staff[stackHead] = newStaff;
		staffN++;
	}
	void pop()
	{
		if (isEmpty())//boþmu
		{
			cout << "Stack is empty !" << endl;
			
		}
		stackHead--;//Tek yapmamýz gereken bir azaltmak
		staffN--;
	}
	const int&top()//Sýradaki elemaný görmek isteyebilir.Çýkarma islemi yapmadan.
	{
		if (isEmpty())
		{
			cout << "Stack is empty!" << endl;
			
		}
		return staff[stackHead];
	}
	void makeEmpty()//Yýðýt ile iþ bittiðinde yýðýtý ilk haline çevirip belleðe geri iade eder.
	{
		stackHead = -1;
		staffN = 0;
		capacity = 0;
		if (staff != NULL)
			delete[] staff;
	}
	~Merhabastack(){
		makeEmpty();
	}
};
#endif
